}

\plot{Figure678_r1}{width=6cm}{
	{Application to synthetic data. 
	(a) Residuals between the simulated data (Figure \ref{fig:synthetic_data}) and 
	predicted data produced by the fast equivalent-layer technique \citep{siqueira-etal2017}. 
	The mean and standard deviation are $8.264 \times 10^{-7}$ and $0.0144$ mGal, respectively.
	(b) Residuals between the simulated data (Figure \ref{fig:synthetic_data}) and 
	predicted data produced by our method. The mean and standard deviation are 
	$8.264 \times 10^{-7}$ and $0.0144$ mGal, respectively.
	(c) Difference between the predicted data produced by the fast equivalent-layer technique 
	and our method.}
	\label{fig:comparison_datafits}
}

\plot{Figure10_r1}{width=6cm}{
	{Application to synthetic data.
	(a) Gravity data produced by the synthetic sources at $z = -300$ m.
	(b) Residuals between the simulated data shown in a and the upward-continued 
	gravity data obtained with the fast equivalent-layer technique (not shown), with 
	mean $-0.0003$ mGal and standard deviation $0.0124$ mGal. 
	(c) Residuals between the simulated data shown in a and the upward-continued 
	gravity data obtained with our method (not shown), with mean $-0.0003$ mGal and 
	standard deviation $0.0124$ mGal.

	our method were $7.62026$ and $0.00834$ seconds, respectively.}
